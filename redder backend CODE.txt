            add: function(id) { this.state.cart.push(this.state.products.find(x => x.id === id)); this.updateBadges(); this.toast('Added to Bag'); },
            remove: function(i) { this.state.cart.splice(i, 1); this.renderCart(document.getElementById('app-root')); this.updateBadges(); },
            toggleWish: function(id) {
                const idx = this.state.wishlist.findIndex(x => x.id === id);
                if(idx === -1) { this.state.wishlist.push(this.state.products.find(x => x.id === id)); this.toast('Saved'); }
                else { this.state.wishlist.splice(idx, 1); this.toast('Removed'); }
                this.updateBadges();
            },
            login: function() { this.state.user = { name: "Guest" }; document.getElementById('auth-status').innerText = "Hi, Guest"; this.router('home'); this.toast('Welcome Back'); },
            applyCode: function() { 
                if(document.getElementById('code').value === 'REDDER10') { this.state.coupon = true; this.renderCart(document.getElementById('app-root')); this.toast('Code Applied'); } 
                else alert('Invalid'); 
            },
            updateBadges: function() { 
                document.getElementById('cart-badge').innerText = this.state.cart.length; 
                document.getElementById('cart-badge').style.display = this.state.cart.length ? 'flex' : 'none';
                document.getElementById('wish-badge').innerText = this.state.wishlist.length;
                document.getElementById('wish-badge').style.display = this.state.wishlist.length ? 'flex' : 'none';
            },
            toast: function(m) { 
                const t = document.getElementById('toast'); t.innerText = m; t.style.visibility = 'visible'; t.className = 'show fixed bottom-10 left-1/2 transform -translate-x-1/2 bg-gray-900 text-white px-6 py-3 rounded shadow-lg text-sm z-50';
                setTimeout(() => t.style.visibility = 'hidden', 2000); 
            }
        };

        // START APP
        app.init();
    </script>
